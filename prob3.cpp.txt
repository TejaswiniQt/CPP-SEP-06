// Swap the value of two integers

#include <iostream>

using namespace std;

void swap(int *a,int *b)
{
    *a = *a ^ *b;
    *b = *a ^ *b;
    *a = *a ^ *b;
}

int main()
{
    int a = 10, b = 15;
    cout<<"Before swapping:\na = "<<a<<" b = "<<b<<endl;
    swap(&a,&b);
    cout<<"After swapping:\na = "<<a<<" b = "<<b<<endl;
    return 0;
}
